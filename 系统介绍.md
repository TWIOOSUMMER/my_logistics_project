# 物流信息管理系统

基于 Django + MySQL + Bootstrap 5 构建的物流仓库管理系统，支持货物管理、订单跟踪、数据可视化与批量操作。

## 技术栈

| 层级 | 技术选型 |
|------|---------|
| 后端框架 | Django 6.0 (Python 3.12) |
| 数据库 | MySQL 8.0 + PyMySQL 驱动 |
| 前端 | Bootstrap 5 + Chart.js 4.4 + Bootstrap Icons 1.11 |
| 字体 | Inter (Google Fonts) + PingFang SC / Microsoft YaHei |
| 模板 | Django Template Language |

class GoodsTb(models.Model):

### 订单表 (order_tb01)

```python
class OrderTb(models.Model):
    order_number = models.CharField(max_length=30, unique=True, verbose_name='订单编号')
    customer_name = models.CharField(max_length=30, verbose_name='客户名称')
    goods = models.ForeignKey(GoodsTb, on_delete=models.CASCADE, verbose_name='关联货物')
    quantity = models.IntegerField(default=1, verbose_name='数量')
    total_amount = models.DecimalField(max_digits=12, decimal_places=2, default=0.00, verbose_name='订单金额')
    status = models.CharField(max_length=20, choices=ORDER_STATUS, default='pending', verbose_name='订单状态')
    created_at = models.DateTimeField(auto_now_add=True, verbose_name='创建时间')
```

订单状态枚举: `pending(待处理)` → `shipping(运输中)` → `delivered(已签收)` / `cancelled(已取消)`

## 核心功能与关键代码

### 1. 运营仪表盘

首页展示 6 张统计卡片、订单状态进度条、客户下单排行、最近操作日志。

```python
def _dashboard():
    return {
        'total_goods': GoodsTb.objects.count(),
        'total_orders': OrderTb.objects.count(),
        'pending_orders': OrderTb.objects.filter(status='pending').count(),
        'total_value': GoodsTb.objects.aggregate(t=Sum(F('price') * F('stock')))['t'] or 0,
        'total_revenue': OrderTb.objects.aggregate(s=Sum('total_amount'))['s'] or 0,
        'low_stock_count': GoodsTb.objects.filter(stock__lte=5).count(),
        'top_customers': OrderTb.objects.values('customer_name').annotate(cnt=Count('id')).order_by('-cnt')[:5],
        'recent_logs': OperationLog.objects.order_by('-created_at')[:8],
    }
```

库存总值通过 `F('price') * F('stock')` 在数据库层聚合计算，无需遍历。

### 2. Chart.js 数据可视化

货物类别分布（环形图） + 近 7 天订单趋势（柱状图）。

```python
def _recent_week_data():
    today = timezone.localdate()
    data = []
    for i in range(6, -1, -1):
        d = today - timedelta(days=i)
        day_start = timezone.make_aware(datetime.combine(d, dt_time.min))
        day_end = timezone.make_aware(datetime.combine(d, dt_time.max))
        cnt = OrderTb.objects.filter(
            created_at__gte=day_start, created_at__lte=day_end
        ).count()
        data.append({'date': d.strftime('%m-%d'), 'count': cnt})
    return data
```

使用 `timezone.make_aware()` 构造时区感知的时间范围，确保与 MySQL 的 UTC 存储对齐。

### 3. 订单金额自动计算

新增/编辑订单时自动计算 `amount = quantity × goods.price`：

```python
def order_add(request):
    if request.method == 'POST':
        form = OrderForm(request.POST)
        if form.is_valid():
            obj = form.save(commit=False)
            obj.total_amount = obj.quantity * obj.goods.price
            obj.save()
            _log('create', '订单', obj.order_number)
            messages.success(request, '订单添加成功')
            return redirect('index')
```

### 4. 列排序

表头点击切换升降序，箭头指示当前排序方向。Django 层面通过 QuerySet 的 `order_by()` 实现：

```python
def _get_sort(request, default, field_map):
    sort = request.GET.get('sort', default)
    reverse = sort.startswith('-')
    key = sort[1:] if reverse else sort
    if key not in field_map:
        sort = default
    return sort

# 视图中使用
goods_qs = GoodsTb.objects.all().order_by(goods_sort)
order_qs = OrderTb.objects.select_related('goods').all().order_by(order_sort)
```

前端通过 JavaScript 切换 URL 参数实现无刷新排序。

### 5. 批量操作

订单列表支持多选 → 一键批量标记状态：

```python
def order_batch_status(request):
    if request.method == 'POST':
        ids = request.POST.getlist('order_ids')
        new_status = request.POST.get('status', '')
        if ids and new_status:
            count = OrderTb.objects.filter(id__in=ids).count()
            OrderTb.objects.filter(id__in=ids).update(status=new_status)
            status_label = dict(OrderTb.ORDER_STATUS).get(new_status, new_status)
            messages.success(request, f'已将 {count} 条订单标记为「{status_label}」')
    return redirect('index')
```

### 6. 级联删除保护

删除货物时检测关联订单数量，前端给出警告。`on_delete=models.CASCADE` 确保删除货物时自动级联删除关联订单：

```python
def goods_delete(request, goods_id):
    goods = get_object_or_404(GoodsTb, id=goods_id)
    related_orders = OrderTb.objects.filter(goods=goods).count()
    if request.method == 'POST':
        name = goods.goods_name
        goods.delete()
        _log('delete', '货物', name, f'删除货物及{related_orders}条关联订单')
        messages.success(request, '货物删除成功')
        return redirect('index')
    return render(request, 'logistics/confirm_delete.html', {
        'name': goods.goods_name,
        'cancel_url': 'index',
        'related_count': related_orders,
    })
```

PAGE_SIZES = [5, 10, 20]

def _get_page_size(request):
    ps = request.GET.get('page_size', str(PAGE_SIZE))
    try:
        ps = int(ps)
    except (ValueError, TypeError):
        ps = PAGE_SIZE
    if ps not in PAGE_SIZES:
        ps = PAGE_SIZE
    return ps

### 9. 表单校验

重量 > 0、单价 > 0、库存 ≥ 0、数量 > 0：

```python
class GoodsForm(forms.ModelForm):
    def clean_weight(self):
        weight = self.cleaned_data.get('weight')
        if weight is not None and weight <= 0:
            raise forms.ValidationError('重量必须大于 0')
        return weight

    def clean_price(self):
        price = self.cleaned_data.get('price')
        if price is not None and price <= 0:
            raise forms.ValidationError('单价必须大于 0')
        return price

    def clean_stock(self):
        stock = self.cleaned_data.get('stock')
        if stock is not None and stock < 0:
            raise forms.ValidationError('库存数量不能为负数')
        return stock
```

### 10. CSV 导入导出

**导出**：直接将 QuerySet 写入 CSV Response：

```python
def goods_export(request):
    response = HttpResponse(content_type='text/csv; charset=utf-8-sig')
    response['Content-Disposition'] = 'attachment; filename="goods_export.csv"'
    writer = csv.writer(response)
    writer.writerow(['ID', '货物名称', '类别', '重量(kg)', '单价(元)', '库存数量'])
    for g in GoodsTb.objects.all().order_by('id'):
        writer.writerow([g.id, g.goods_name, g.category, g.weight, g.price, g.stock])
    return response
```

**导入**：读取上传的 CSV 文件，逐行创建货物记录：

```python
def goods_import(request):
    if request.method == 'POST':
        csv_file = request.FILES.get('csv_file')
        decoded = csv_file.read().decode('utf-8-sig')
        reader = csv.reader(io.StringIO(decoded))
        next(reader)  # 跳过表头
        for row in reader:
            GoodsTb.objects.create(
                goods_name=row[0].strip(),
                category=row[1].strip() or '未分类',
                weight=float(row[2]) if row[2] else 0,
                price=float(row[3]) if row[3] else 0,
                stock=int(float(row[4])) if row[4] else 0,
            )
```
